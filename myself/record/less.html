    # less常用

    # 变量定义

       @primary-color: #3498db;
   body {
   color: @primary-color;
   }

    # 嵌套规则

         nav {
        ul {
           margin: 0;
           padding: 0;
           list-style: none;
        }
        li { display: inline-block; }
     }

    # 混合（Mixin）

      .border-radius(@radius) {
      border-radius: @radius;
     -webkit-border-radius: @radius;
     -moz-border-radius: @radius;
  }

  .button {
     .border-radius(5px);
  }

    # 运算

       @base: 5%;
   @width: @base * 2

    # 条件语句

       @theme: light;
   body {
   color: if(@theme == light, #000, #fff);
   }

    # 循环

       .generate-columns(@n) {
   .column {
      width: (@n * 100%) / 12;
   }
   }

   .generate-columns(3);

    # 命名空间

       #header {
   color: black;
   }
   #footer {
   color: red;
   }

    # 函数颜色

       @color: #336699;
   .lighter {
   color: lighten(@color, 20%); // 使颜色变亮
   }
